# U-Boot saved environment flashed at MMC offset 14 MiB.
#
# NOTE: U-Boot's saved env is a wholesale replacement for the compiled-in
# default env, not a merge. Anything U-Boot relies on to autoboot (bootcmd,
# bootdelay, address map) MUST be duplicated here or autoboot drops to prompt.

# Boot configuration (duplicates CONFIG_BOOTCOMMAND / distro defaults)
bootdelay=2
bootcmd=if test "${boot_slot}" = "B"; then setenv bootpart 3; else setenv bootpart 1; fi; load mmc 0:${bootpart} ${scriptaddr} boot.scr && source ${scriptaddr}

# Standard aarch64 RK3566 load addresses (normally provided by distro defaults)
scriptaddr=0x00c00000
kernel_addr_r=0x02000000
fdt_addr_r=0x12000000

# A/B rollback state
boot_slot=A
bootcount=0
bootlimit=3
upgrade_available=0
altbootcmd=if test "${upgrade_available}" = "1"; then if test "${boot_slot}" = "A"; then setenv boot_slot B; else setenv boot_slot A; fi; setenv upgrade_available 0; fi; setenv bootcount 0; saveenv; run bootcmd
